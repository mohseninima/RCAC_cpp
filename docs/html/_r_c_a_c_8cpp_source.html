140 void RCAC::computeFiltered()
141 {
142  //Loop through and filter the coefficients
143  Eigen::MatrixXd ufSum = Eigen::MatrixXd::Zero(lz, 1);
144  Eigen::MatrixXd PhifSum = Eigen::MatrixXd::Zero(lz, Nc*lu*(lu+ly));
145 
146  //Filter the numerator coefficients
147  //for (int i = 0; i < FILT.filtNu.cols()/lu; i++)
148  for (int i = 0; i < filtorder; i++)
149  {
150  ufSum = FILT.filtNu.block(0, lu*i, lz, lu)*uBar[i] + ufSum;
151  PhifSum = FILT.filtNu.block(0, lu*i, lz, lu)*PhiBar[i] + PhifSum;
152  }
153 
154  //Filter the Denominator coefficients coefficients
155  for (int i = 0; i < (filtorder-1); i++)
156  {
157  ufSum = ufSum - FILT.filtDu.block(0, lz*i, lz, lz)*ufBar[i];
158  PhifSum = PhifSum - FILT.filtDu.block(0, lz*i, lz, lz)*PhifBar[i];
159  }
160 
161  //Add the sum to the list of past filtered values for future filtering
162  ufBar.push_front(ufSum);
163  PhifBar.push_front(PhifSum);
164  //remove the oldest filtered value
165  ufBar.pop_back();
166  PhifBar.pop_back();
167 }
