1 Getting Started {#GettingStarted}
2 ===============
3 This is a guide on getting started with the RCAC library.
4 
5 How to install?
6 ===============
7 First download the source code from the [repository](https://github.com/mohseninima/RCAC_cpp).
8 Only the *.cpp and *.hpp files are needed. Next, you will need a copy of the Eigen matrix library,
9 located [here](http://eigen.tuxfamily.org/). Extract the *Eigen* and *unsupported* folders and place them in the same folder as the RCAC code.
10 You should have a folder structure similar to the following.
11 
12 ~~~~~~~~~~~~~~~~~~~
13 <root>
14  Eigen/
15  unsupported/
16  RCAC.cpp
17  RCAC.hpp
18  RCACCreator.hpp
19  RCAC*.cpp
20  RCAC*.hpp
21  .
22  .
23  .
24  main.cpp (your code)
25 ~~~~~~~~~~~~~~~~~~~
26 
27 Using Recursive Least Squares (RLS) RCAC in C++
28 =========================================
29 Using RLS RCAC in your own code requires 3 main steps
30 
31 1. Including "RCACCreator.hpp" and Eigen
32 2. Defining and assigning the rcacRlsFlags struct
33 3. Defining and assigning the rcacFilt struct
34 4. Initializing RCAC
35 5. Using the oneStep() method to propagate RCAC by one step
36 6. Using the getControl() method to get the associated control input
37 
38 ~~~~~~~~~~~~~~~~~~~{.cpp}
39 #include "Eigen/Core"
40 #include "RCACCreator.hpp"
41 #include <iostream>
42 #include <string>
43 using namespace Eigen;
44 
45 int main()
46 {
47  //Initialize the plant
48  int lx = 2;
49  int lu = 2;
50  int ly = 2;
51  int lz = ly;
52 
53  MatrixXd A(lx,lx);
54  A << 0.25,1,
55  0,0.25;
56 
57  MatrixXd B(lx,lu);
58  B << 1,0,
59  0,1;
60 
61  MatrixXd C(ly,lx);
62  C << 1,0,
63  0,3;
64 
65  MatrixXd x(lx,1);
66  x << 0,
67  0;
68 
69  //End time of the simulation
70  int kend = 30;
71 
72  //Set Flags
73  rcacRlsFlags FLAGS;
74  FLAGS.lu = lu;
75  FLAGS.lz = lz;
76  FLAGS.ly = ly;
77  FLAGS.Nc = 4;
78  int ltheta = FLAGS.Nc*FLAGS.lu*(FLAGS.lu+FLAGS.ly);
79  FLAGS.filtorder = 4;
80  FLAGS.k_0 = FLAGS.Nc;
81  FLAGS.P0 = 100000*MatrixXd::Identity(ltheta, ltheta);
82  FLAGS.Ru = MatrixXd::Zero(FLAGS.lu, FLAGS.lu);
83  FLAGS.Rz = MatrixXd::Identity(FLAGS.lz, FLAGS.lz);
84  FLAGS.lambda = 1;
85  FLAGS.theta_0 = MatrixXd::Zero(ltheta, 1);
86 
87  std::cout << "Flags Set\n";
88 
89  //Set Filter
90  rcacFilt FILT;
91  int Hnum = FLAGS.filtorder-1;
92  FILT.filtNu.resize(FLAGS.lz, FLAGS.lu*(Hnum+1));
93  FILT.filtDu.resize(FLAGS.lz, FLAGS.lz*Hnum);
94  //FILT.filtNu << 0, 1, 0.25, 0.0625;
95  FILT.filtNu << 0,0, 1,0, 0.25,1, 0.0625,0.5,
96  0,0, 0,3, 0,0.75, 0,0.1875;
97  //FILT.filtNu << 0, 0, 1, 2, 0.25, 0.5, 0.0625, 0.125;
98  //FILT.filtDu << 0, 0, 0;
99  FILT.filtDu << 0, 0, 0, 0, 0, 0,
100  0, 0, 0, 0, 0, 0;
101  FILT.filtNz = MatrixXd::Identity(FLAGS.lz, FLAGS.lz);
102  FILT.filtDz = MatrixXd::Zero(FLAGS.lz, FLAGS.lz);
103 
104  std::cout << "Filter Set\n";
105 
106  //Initialize RCAC
107  std::string rcacType = "RLS";
108  RCAC *myRCAC;
109  //myRCAC = RCAC::init<rcacRlsFlags>(FLAGS, FILT, whichRcac);
110  //Call the factory method to initialize RCAC
111  myRCAC = myRCAC->init<rcacRlsFlags>(FLAGS, FILT, rcacType);
112  std::cout << "RLS Init\n";
113 
114  //Run Simulation
115  VectorXd y(FLAGS.ly,1);
116  VectorXd z(FLAGS.lz,1);
117  VectorXd u(FLAGS.lu,1);
118  u = MatrixXd::Zero(FLAGS.lu, 1);
119 
120  for (int k = 0; k < kend; k++)
121  {
122  y = C*x;
123  z = y - VectorXd::Ones(FLAGS.ly);
124 
125  x = A*x + B*u;
126 
127  myRCAC->oneStep(u, z, z);
128  u = myRCAC->getControl();
129 
130  //std::cout << "y: " << y << ", z: " << z << ", u: " << u << std::endl;
131  }
132  return(0);
133 }
